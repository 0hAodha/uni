Intbuffer question answer:

public class IntBufferImp {
    private int values;
    private boolean available = false;

    public int consume() {
        if (!available) {
            try {
                wait();
            }
            catch (InterruptedException e) {
                e.printStackTrace();
            }
        }

        available = false;
        notifyAll();
        return value;
    }

    public void produce(int i) {
        if (available) {
            try {
                wait();
            }
            catch (InterruptedException e) {
                e.printStackTrace();
            }
        }

        value = i;
        notifyAll();
        available = true;
    }
}
